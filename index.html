Ajax Request Exemplo


var xhr = new XMLHttpRequest();
xhr.open('GET', 'api.php?name=joao');

xhr.onreadystatechange = function () {
	var DONE = 4; // readyState 4 means the request is done
	var OK = 200; // status 200 is a successful return
	if (xhr.readyState === DONE) {
		if (xhr.status === OK) {
			console.log(xhr.responseText);
		} else {
			console.log('Error: ' + xhr.status);
		}
	}
};
xhr.send(null);

ES5 - Herança


/* Father Class BEGIN */
//construct
function Father(){
	console.log('Father executed!');
}
//properties
Father.prototype.fathersProp = "Fathers";
//methods
Father.prototype.fathersMethod = function(){
	console.log("Fathers Method!");
}
/* Father Class END */
					
/* Child Class BEGIN */
//construct
function Child(){
  Father.apply(this, arguments); //super
  console.log('Child executed!');
}
//heritage
Child.prototype = Object.create(Father.prototype);
//properties
Child.prototype.childsProp = "Childs";
//methods
Child.prototype.childsMethod = function(){
  console.log("Childs Methods!");
}
/* Child Class END */

Anonymous Closure


var nota = 4;

(function(){

	nota = 6;

	var elevar = function(){
		return ++nota;
	}

	console.log(elevar(nota));

})();

console.log(nota);

Global Import


var nota = 4;

(function(globalVar){

	var elevar = function(){
		globalVar += 1;
		return globalVar;
	}

	console.log(elevar(globalVar));

})(nota);

console.log(nota);

Object Interface


var sis = (function(){
	//private
	var nota = 10;
	var getNota = function(){return nota;}

	//exported
	return {
		media: 19,
		showMedia: function(){
			console.log(this.media);
		}
	}
})();

sis.showMedia();

Revealing Module Pattern


var sis = (function(){
	var nota = 10;
	var media = 19;
	var getNota = function(){ return nota; }
	var showMedia = function(){
		console.log(this.media);
	}

	return {
		media: media,
		showMedia: showMedia
	}
})();

sis.showMedia();

CommonJS


//Arquivo menu.js
function menuModule {
	this.render = function(){
	  console.log('rendering...');
	}
}
module.exports = menuModule;

//Arquivo index.js
var vertical = require('menu');
vertical.render();

AMD


define(['menu','sidebar'], function(menu, sidebar){
	var init = function(){
		menu.render();
		sidebar.render();
	}

	return {
		init: init
	}
});
